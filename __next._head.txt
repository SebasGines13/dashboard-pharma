// Datos mockup para el Tablero de Decisión — Empaque Farmacéutico Digitalizado (PharmaXXX)
// Basado en el caso "Digitalización del proceso de empaque en industria farmacéutica" (Grupo 10)

export type EstadoOrden = "En ejecución" | "En revisión" | "Liberado" | "Detenido" | "Programado"
export type Severidad = "Crítica" | "Mayor" | "Menor"

export type EstadoOEE = "Operativa" | "Atención" | "Detenida"

export interface Linea {
  id: string
  nombre: string
  producto: string
  lote: string
  estado: EstadoOrden
  avance: number // %
  excepcionesAbiertas: number
  reconciliacionPendiente: boolean
  tiempoLiberacionEstimado: string
  operador: string
  // OEE = Disponibilidad x Rendimiento x Calidad
  oee: number // %
  disponibilidad: number // %
  rendimiento: number // %
  calidad: number // % (First Pass Yield)
  velocidad: number // unidades/min
}

export interface Firma {
  id: string
  documento: string
  paso: string
  lote: string
  linea: string
  responsable: string
  rol: string
  vence: string // hora límite
  urgente: boolean
}

export type TipoAlerta = "Crítica" | "Advertencia" | "Info"

export interface Alerta {
  id: string
  tipo: TipoAlerta
  titulo: string
  descripcion: string
  linea: string
  haceMin: number
  resuelta: boolean
}

export interface Excepcion {
  id: string
  linea: string
  lote: string
  descripcion: string
  severidad: Severidad
  hora: string
  estado: "Abierta" | "En análisis" | "Resuelta"
}

export interface Lote {
  id: string
  producto: string
  linea: string
  cantidad: number
  estado: EstadoOrden
  inicio: string
  reconciliacion: number // % de materiales reconciliados
}
