"use client"

import {
  createContext,
  useCallback,
  useContext,
  useEffect,
  useMemo,
  useRef,
  useState,
} from "react"
import {
  lineas as lineasSeed,
  excepciones as excepcionesSeed,
  lotesEnRevision as lotesSeed,
  firmas as firmasSeed,
  alertas as alertasSeed,
  produccionPorHora,
  tiempoLiberacion,
  oeeHistorico,
  resumenTurno,
  type Linea,
  type Excepcion,
  type Lote,
  type Firma,
  type Alerta,
  type EstadoOrden,
  type EstadoOEE,
} from "@/lib/mock-data"

type FiltroEstado = EstadoOrden | "Todos"
type Tema = "light" | "dark"

export function estadoOEE(linea: Linea): EstadoOEE {
  if (linea.estado === "Detenido") return "Detenida"
  if (linea.oee > 0 && linea.oee < 75) return "Atención"
  return "Operativa"
}

interface DashboardState {
  lineas: Linea[]
  excepciones: Excepcion[]
  lotes: Lote[]
  firmas: Firma[]
  alertas: Alerta[]
  // tema
  tema: Tema
  toggleTema: () => void
  // filtros
  busqueda: string
  filtroEstado: FiltroEstado
  setBusqueda: (v: string) => void
  setFiltroEstado: (v: FiltroEstado) => void
  resetFiltros: () => void
  lineasFiltradas: Linea[]
  // modo en vivo
  enVivo: boolean
  toggleEnVivo: () => void
  ultimaActualizacion: Date
  // acciones
  avanzarExcepcion: (id: string) => void
  resolverExcepcion: (id: string) => void
  liberarLinea: (id: string) => void
  detenerLinea: (id: string) => void
  reanudarLinea: (id: string) => void
  firmar: (id: string) => void
  resolverAlerta: (id: string) => void
  descartarAlerta: (id: string) => void
  // derivados
  kpis: {
    ordenesActivas: number
    excepcionesAbiertas: number
    lotesEnRevision: number
    reconciliacionAutomatica: number
    cumplimientoGMP: number
    tiempoLiberacionPromedio: number
    oeeGlobal: number
    firmasPendientes: number
    alertasActivas: number
  }
  distribucionEstados: { estado: string; valor: number; fill: string }[]
  incidenciaPorLinea: { linea: string; excepciones: number }[]
  produccionPorHora: typeof produccionPorHora
  tiempoLiberacion: typeof tiempoLiberacion
  oeeHistorico: typeof oeeHistorico
  resumenTurno: typeof resumenTurno
}

const DashboardContext = createContext<DashboardState | null>(null)

const FILL_ESTADOS: Record<string, string> = {
  "En ejecución": "var(--chart-1)",
  "En revisión": "var(--chart-3)",
  Liberado: "var(--chart-5)",
  Detenido: "var(--destructive)",
  Programado: "var(--muted-foreground)",
}

export function DashboardProvider({ children }: { children: React.ReactNode }) {
  const [lineas, setLineas] = useState<Linea[]>(() => lineasSeed.map((l) => ({ ...l })))
  const [excepciones, setExcepciones] = useState<Excepcion[]>(() => excepcionesSeed.map((e) => ({ ...e })))
  const [lotes] = useState<Lote[]>(() => lotesSeed.map((l) => ({ ...l })))
  const [firmas, setFirmas] = useState<Firma[]>(() => firmasSeed.map((f) => ({ ...f })))
  const [alertas, setAlertas] = useState<Alerta[]>(() => alertasSeed.map((a) => ({ ...a })))
  const [tema, setTema] = useState<Tema>("light")
  const [busqueda, setBusqueda] = useState("")
  const [filtroEstado, setFiltroEstado] = useState<FiltroEstado>("Todos")
  const [enVivo, setEnVivo] = useState(false)
  const [ultimaActualizacion, setUltimaActualizacion] = useState<Date>(() => new Date())
  const tickRef = useRef(0)

  // Aplica la clase .dark al <html> para que los tokens de globals.css cambien
  useEffect(() => {
    const root = document.documentElement
    if (tema === "dark") root.classList.add("dark")
    else root.classList.remove("dark")
  }, [tema])

  const toggleTema = useCallback(() => setTema((t) => (t === "dark" ? "light" : "dark")), [])

  const resetFiltros = useCallback(() => {
    setBusqueda("")
    setFiltroEstado("Todos")
  }, [])

  const toggleEnVivo = useCallback(() => setEnVivo((v) => !v), [])

  // Simulación "en vivo": avanza el progreso de las líneas y fluctúa el OEE
  useEffect(() => {
    if (!enVivo) return
    const intervalo = setInterval(() => {
      tickRef.current += 1
      setUltimaActualizacion(new Date())
      setLineas((prev) =>
        prev.map((l) => {
          if (l.estado === "Detenido" || l.estado === "Programado" || l.estado === "Liberado") return l
          // fluctuación leve del OEE para líneas activas
          const jitter = (n: number, max = 100) =>
            Math.max(0, Math.min(max, Math.round(n + (Math.random() * 4 - 2))))
          const disponibilidad = jitter(l.disponibilidad)
          const rendimiento = jitter(l.rendimiento)
          const calidad = jitter(l.calidad)
          const oee = Math.round((disponibilidad * rendimiento * calidad) / 10000)
          if (l.estado !== "En ejecución") {
            return { ...l, disponibilidad, rendimiento, calidad, oee }
          }
          const incremento = Math.round(2 + Math.random() * 5)
          const nuevoAvance = Math.min(100, l.avance + incremento)
          return {
            ...l,
            avance: nuevoAvance,
            estado: nuevoAvance >= 100 ? "En revisión" : l.estado,
            disponibilidad,
            rendimiento,
            calidad,
            oee,
            velocidad: Math.max(0, Math.round(l.velocidad + (Math.random() * 20 - 10))),
            tiempoLiberacionEstimado:
              nuevoAvance >= 100
                ? "1.0 h"
                : `${Math.max(0.5, (100 - nuevoAvance) / 25).toFixed(1)} h`,
          }
        }),
      )
    }, 1800)
    return () => clearInterval(intervalo)
  }, [enVivo])

  const avanzarExcepcion = useCallback((id: string) => {
    setExcepciones((prev) =>
      prev.map((e) =>
        e.id === id ? { ...e, estado: e.estado === "Abierta" ? "En análisis" : "Resuelta" } : e,
      ),
    )
  }, [])

  const resolverExcepcion = useCallback((id: string) => {
    setExcepciones((prev) => prev.map((e) => (e.id === id ? { ...e, estado: "Resuelta" } : e)))
  }, [])

  const liberarLinea = useCallback((id: string) => {
    setLineas((prev) =>
      prev.map((l) =>
        l.id === id
          ? {
              ...l,
              estado: "Liberado",
              avance: 100,
              tiempoLiberacionEstimado: "Liberado",
              reconciliacionPendiente: false,
            }
          : l,
      ),
    )
  }, [])

  const detenerLinea = useCallback((id: string) => {
    setLineas((prev) => prev.map((l) => (l.id === id ? { ...l, estado: "Detenido", velocidad: 0 } : l)))
  }, [])

  const reanudarLinea = useCallback((id: string) => {
    setLineas((prev) => prev.map((l) => (l.id === id ? { ...l, estado: "En ejecución" } : l)))
  }, [])

  const firmar = useCallback((id: string) => {
    setFirmas((prev) => prev.filter((f) => f.id !== id))
  }, [])

  const resolverAlerta = useCallback((id: string) => {
    setAlertas((prev) => prev.map((a) => (a.id === id ? { ...a, resuelta: true } : a)))
  }, [])

  const descartarAlerta = useCallback((id: string) => {
    setAlertas((prev) => prev.filter((a) => a.id !== id))
  }, [])

  const lineasFiltradas = useMemo(() => {
    const q = busqueda.trim().toLowerCase()
    return lineas.filter((l) => {
      const coincideEstado = filtroEstado === "Todos" || l.estado === filtroEstado
      const coincideBusqueda =
        q === "" ||
        l.nombre.toLowerCase().includes(q) ||
        l.producto.toLowerCase().includes(q) ||
        l.lote.toLowerCase().includes(q) ||
        l.operador.toLowerCase().includes(q)
      return coincideEstado && coincideBusqueda
    })
  }, [lineas, busqueda, filtroEstado])

  const kpis = useMemo(() => {
    const ordenesActivas = lineas.filter((l) => l.estado === "En ejecución").length
    const excepcionesAbiertas = excepciones.filter((e) => e.estado !== "Resuelta").length
    const lotesEnRevision = lineas.filter((l) => l.estado === "En revisión").length
    const tiempos = tiempoLiberacion.map((t) => t.horas)
    const tiempoLiberacionPromedio = tiempos.reduce((a, b) => a + b, 0) / tiempos.length
    const activas = lineas.filter((l) => l.oee > 0)
    const oeeGlobal = activas.length
      ? Math.round(activas.reduce((a, l) => a + l.oee, 0) / activas.length)
      : 0
    return {
      ordenesActivas,
      excepcionesAbiertas,
      lotesEnRevision,
      reconciliacionAutomatica: 96.4,
      cumplimientoGMP: 99.2,
      tiempoLiberacionPromedio,
      oeeGlobal,
      firmasPendientes: firmas.length,
      alertasActivas: alertas.filter((a) => !a.resuelta).length,
    }
  }, [lineas, excepciones, firmas, alertas])

  const distribucionEstados = useMemo(() => {
    const conteo: Record<string, number> = {}
    for (const l of lineas) conteo[l.estado] = (conteo[l.estado] ?? 0) + 1
    return Object.entries(conteo)
      .map(([estado, valor]) => ({ estado, valor, fill: FILL_ESTADOS[estado] ?? "var(--muted-foreground)" }))
      .sort((a, b) => b.valor - a.valor)
  }, [lineas])

  const incidenciaPorLinea = useMemo(
    () => lineas.map((l) => ({ linea: l.id, excepciones: l.excepcionesAbiertas })),
    [lineas],
  )

  const value: DashboardState = {
    lineas,
    excepciones,
    lotes,
    firmas,
    alertas,
    tema,
    toggleTema,
    busqueda,
    filtroEstado,
    setBusqueda,
    setFiltroEstado,
    resetFiltros,
    lineasFiltradas,
    enVivo,
    toggleEnVivo,
    ultimaActualizacion,
    avanzarExcepcion,
    resolverExcepcion,
    liberarLinea,
    detenerLinea,
    reanudarLinea,
    firmar,
    resolverAlerta,
    descartarAlerta,
    kpis,
    distribucionEstados,
    incidenciaPorLinea,
    produccionPorHora,
    tiempoLiberacion,
    oeeHistorico,
    resumenTurno,
  }

  return <DashboardContext.Provider value={value}>{children}</DashboardContext.Provider>
}

export function useDashboard() {
  const ctx = useContext(DashboardContext)
  if (!ctx) throw new Error("useDashboard debe usarse dentro de DashboardProvider")
  return ctx
}
